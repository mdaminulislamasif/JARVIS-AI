if result['status'] == 'success':
    print(f"\n📊 Test Results:")
    print(f"  Tests Run: {result['tests_run']}")
    print(f"  Passed: {result['tests_passed']}")
    print(f"  Failed: {result['tests_failed']}")

        print("\n✅ All tests passed!")
    else:
        print(f"\n⚠️ {result['tests_failed']} tests failed")
else:
    print("⚠️ No tests found or testing failed")

⚡ STEP 5: COMPLETE CYCLE / সম্পূর্ণ Cycle
═══════════════════════════════════════════════════════════════════════════════

CREATE FILE: complete_cycle.py
───────────────────────────────────────────────────────────────────────────────
```python
from jarvis_development_master import DevelopmentMaster

dev = DevelopmentMaster()

project_path = '.'
language = 'python'

print("="*60)
print("COMPLETE DEVELOPMENT CYCLE")
print("="*60)

# 1. Learn
print("\n1. LEARNING...")
dev.learn_to_build(language)
dev.learn_to_test(language)
dev.learn_to_operate('all')

# 2. Build
print("\n2. BUILDING...")
build_result = dev.build_project(project_path, language)
print(f"Build status: {build_result['status']}")

# 3. Test
print("\n3. TESTING...")
test_result = dev.test_project(project_path, language)
if test_result['status'] == 'success':
    print(f"Tests: {test_result['tests_passed']}/{test_result['tests_run']} passed")

# 4. Operate
print("\n4. OPERATING...")
op_result = dev.operate_project(project_path, 'monitor')
print(f"Operation status: {op_result['status']}")

# 5. Statistics
print("\n5. STATISTICS:")
stats = dev.get_statistics()
for key, value in stats.items():
    print(f"  {key}: {value}")

print("\n✅ Complete cycle finished!")
```

RUN:
```bash
python complete_cycle.py
```

USE CASE 3: Continuous Testing
───────────────────────────────────────────────────────────────────────────────
```python
from jarvis_development_master import DevelopmentMaster
import time

dev = DevelopmentMaster()

project_path = '.'
language = 'python'

# Run tests every 5 minutes
while True:
    print("\nRunning tests...")
    result = dev.test_project(project_path, language)
    
    if result['status'] == 'success':
        if result['tests_failed'] == 0:
            print("✅ All tests passed!")
        else:
            print(f"⚠️ {result['tests_failed']} tests failed!")
    
    print("Waiting 5 minutes...")
    time.sleep(300)  # 5 minutes
```

🔧 INTEGRATION WITH JARVIS PANEL / JARVIS Panel এ যুক্ত করুন
═══════════════════════════════════════════════════════════════════════════════

ADD TO jarvis_panel.py:
───────────────────────────────────────────────────────────────────────────────

1. Import:
```python
from jarvis_development_master import DevelopmentMaster
```

2. Initialize:
```python
self.dev_master = DevelopmentMaster()
```

3. Add buttons:
```python
# --- MODULE SECTION: DEVELOPMENT ---
ctk.CTkLabel(self.modules, text="[ DEVELOPMENT ]", 
             font=("Courier New", 10, "bold"), 
             text_color="#555555").pack(pady=5)

add_module("🏗️ LEARN BUILD", "learnbuild", "#006600", "#008800")
add_module("💻 LEARN DEVELOP", "learndev", "#000066", "#000088")
add_module("⚙️ LEARN OPERATE", "learnop", "#660000", "#880000")
add_module("🧪 LEARN TEST", "learntest", "#006666", "#008888")
add_module("🏗️ BUILD PROJECT", "buildproj", "#004400", "#006600")
add_module("🧪 TEST PROJECT", "testproj", "#440044", "#660066")
add_module("📊 DEV STATS", "devstats", "#444400", "#666600")
```

4. Add handlers:
```python
elif "learnbuild" in cmd:
    lang = self.entry.get().strip() or 'python'
    self.dev_master.learn_to_build(lang)
    self.log("JARVIS", f"Learned to build {lang}")

elif "learndev" in cmd:
    self.dev_master.learn_to_develop('all')
    self.log("JARVIS", "Learned development practices")

elif "learnop" in cmd:
    self.dev_master.learn_to_operate('all')
    self.log("JARVIS", "Learned operations")

elif "learntest" in cmd:
    lang = self.entry.get().strip() or 'python'
    self.dev_master.learn_to_test(lang)
    self.log("JARVIS", f"Learned to test {lang}")

elif "buildproj" in cmd:
    path = self.entry.get().strip() or '.'
    result = self.dev_master.build_project(path, 'python')
    self.log("JARVIS", f"Build: {result['status']}")

elif "testproj" in cmd:
    path = self.entry.get().strip() or '.'
    result = self.dev_master.test_project(path, 'python')
    if result['status'] == 'success':
        self.log("JARVIS", f"Tests: {result['tests_passed']}/{result['tests_run']} passed")

elif "devstats" in cmd:
    stats = self.dev_master.get_statistics()
    self.log("JARVIS", f"Dev Stats: {stats}")
```
